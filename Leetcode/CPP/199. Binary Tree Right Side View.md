BFS.
Time complexity: O(n).
Space complexity: O(n).

C++ code:
```
class Solution {
public:
    vector<int> rightSideView(TreeNode* root) {
        vector<int> res;
        if (!root) { return res; }
        
        queue<TreeNode*> q;
        q.push(root);
        
        while (q.size() != 0) {
            int size = q.size();
            
            for (int i = 0; i < size; ++i) {
                TreeNode* node = q.front();
                q.pop();
                
                if (node->left) { q.push(node->left); }
                if (node->right) { q.push(node->right); }
                if (i == size - 1) {
                    res.push_back(node->val);
                }
            }
        }
        
        return res;
    }
};
```
